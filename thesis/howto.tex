def StepVerlet(ODESystem, rv, t, dt, aCoulomb, mass, charge, trapParams):

    r, v = rv
    v, a = ODESystem(rv, t, aCoulomb, mass, charge, trapParams)
    
    r1 = r + v * dt + 0.5 * a * dt**2
    
    a1 = ODESystem(np.array([r1, v]), t, aCoulomb, mass, charge, trapParams)[1]    
    
    v1 = v + 0.5 * (a + a1) * dt
    t1 = t + dt
    
    rv1 = np.array([r1, v1])
    
    return rv1, t1   
\end{Verbatim}
